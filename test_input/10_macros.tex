\section*{Macro Tests}

\defanchor{denominator}
\textbf{Denominator}: the bottom part of a fraction.

\questionanchor{34}

\mainStep{1}{Rewrite the denominator $(4-x^2)^{\frac{3}{2}}$.}

\solutionStep{1}{Identify the \qdeflink{denominator}{denominator}}{
In
\[
\frac{x^2}{(4-x^2)^{\frac{3}{2}}},
\]
the denominator is
\[
(4-x^2)^{\frac{3}{2}}.
\]
}

\solutionStep{1}{Use trigonometric substitution}{
When the expression contains
\[
\sqrt{a^2-x^2},
\]
we can let
\[
x=a\sin\theta.
\]
See \backtoterm{denominator}{the linked term}.
}
